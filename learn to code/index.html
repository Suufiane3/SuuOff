    text_rect = text.get_rect(center=(x, y))
    screen.blit(text, text_rect)

    output = [' '] * screen_size
    zbuffer = [0] * screen_size

    for theta in range(0, 628, theta_spacing):

            sintheta = sin(theta)

            sinphi = sin(phi)

            circley = R1 * sintheta


            x = circlex * (cosB * cosphi + sinA * sinB * sinphi) - circley * cosA * sinB
